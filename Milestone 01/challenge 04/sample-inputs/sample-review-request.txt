function calculateTotal(items) {
  let total = 0;
  for (let i = 0; i < items.length; i++) {
    const item = items[i];
    if (item.price > 0) {
      if (item.taxable) {
        total += item.price * 1.15;
      } else {
        total += item.price;
      }
    }
  }
  return total.toFixed(2);
}

// Test call
console.log(calculateTotal([{price: 10, taxable: true}, {price: 5, taxable: false}]));
